42 // Se estiver, encerra o loop pra retornar essa opção.
43 break;
44 } else {
45 // Se não estiver, retorna um erro que sera tratado.

47 }
48 } catch (std::invalid_argument) {
49 // Trata o caso do usuário escolher um caractere
